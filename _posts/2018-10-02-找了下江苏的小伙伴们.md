```python
from urllib import request
from urllib import error
import time

for i in range(1000,1500):
	try:
		f=request.urlopen("https://www.luogu.org/team/show?teamid=" + str(i))
	except error.HTTPError:
		continue
	else:
		data = f.read()
		tmp = data.decode("utf-8")
		if tmp.find("江苏")!=-1:
			print(i)
		#time.sleep(0.1)
```
